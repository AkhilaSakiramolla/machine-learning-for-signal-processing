Activation Function (sigmoid/logistic)¶

In [9]:
def sigmoid(x, derivative = False):
    if derivative:
        if derivative == True:
            return np.exp(-x)/(1+np.exp(-x))**2
        else:
            return 1 / (1 + np.exp(-x))
    else:
        return 1/(1+np.exp(-x))

def perceptron(Y, weights_bias, alpha):
    iteration = 0
    df = pd.DataFrame(columns=['iteration','error'])
    #Y_prime = np.vstack((Y,np.ones((1,Y.shape[1]))))
    
    while iteration <= 300000:
        iteration+=1
        Y_prime = np.vstack((Y,np.ones((1,Y.shape[1])))) # stacking additional row with ones to handle bias

        # Forward propagation
        in_to_out = weights_bias @ Y_prime
        pred = sigmoid(in_to_out)

        # Backward propagation
        delta = (pred - actual) * sigmoid(in_to_out, derivative = True)
        delta1 = delta @ Y_prime.T

        # Update bias and weights
        weights_bias -= alpha * delta1

        # Error calculation
        error = np.sum(0.5 * ((actual-pred)**2))

        #print(iteration, error)
        df.loc[len(df.index)] = [iteration, error] #appending to the error dataframe
    
    return df,pred

Multilayer Perceptron (One hidden layer)¶

Forward Propagation, Backward Propagation, Error Calculation

def multilayer_percepton(X, weights_bias1, weights_bias2, alpha):
    iteration = 0
    #error_list = []
    df = pd.DataFrame(columns=['iteration','error'])
    X1_prime = np.vstack((X,np.ones((1,X.shape[1]))))   # appending a row of ones to handle bias
    #X2_prime = np.vstack((X,np.ones((1,X.shape[1]))))
    while iteration <= 200000: # number of iterations
        
        iteration+=1
        # Forward propagation
        in_to_hid = weights_bias1 @ X1_prime
        init_pred = sigmoid(in_to_hid)
        init_pred = np.vstack((init_pred,np.ones((1,X.shape[1]))))

        hid_to_out = weights_bias2.T @ init_pred
        final_pred = sigmoid(hid_to_out)

        # Backward propagation
        delta_out = (final_pred - actual) * sigmoid(hid_to_out, derivative = True)
        final_delta_out = delta_out @ init_pred.T

        delta_hid = (weights_bias2 @ delta_out)[0:3,:] * sigmoid(in_to_hid, derivative = True)
        final_delta_hid = delta_hid @ X1_prime.T

        # Update bias and weights
        weights_bias1 -= alpha * final_delta_hid
        weights_bias2 -= alpha * final_delta_out.T

        # Error calculation
        error = np.sum(0.5 * ((actual-final_pred)**2))

        df.loc[len(df.index)] = [iteration, error]   # appending to dataframe
    return df,final_pred

# Function that collates and returns the train or test data
def train_or_test_string(digit, strt_num, end_num):
    train_data = []
    
    for n_rec in range(strt_num,end_num):
        string = str(digit) + "_jackson_" + str(n_rec) + '.wav'
        train_data.append(string)
    
    return train_data

# Function to calculate the X and T
def string_function(string_value1,string_value2,num): 
    string = "("
    if string_value2 == ".shape[1]":
        for i in range(num):
            string+= string_value1 + str(i) + "]" + string_value2 + ","
        string = string[0:-1]
        string+=")"
    else:
        for i in range(num):
            string+= string_value1 + str(i) + "],"
        string = string[0:-1]
        string+=")"
        
    return string

# Trains the GMMHMM model
def training(X_,T_):
    h = hmm.GMMHMM(n_components=2, n_mix=1, min_covar=0.001, startprob_prior=1.0, \
                            transmat_prior=1.0, weights_prior=1.0, means_prior=0.0, means_weight=0.0, \
                            covars_prior=None, covars_weight=None, algorithm='viterbi', covariance_type='diag', \
                            random_state=None, n_iter=20, tol=0.01, verbose=False, params='stmcw', \
                            init_params='stmcw', implementation='log')
    
    
    model = h.fit(X_.T,lengths = T_)   # fit the model
    return model

    model_dict[digit] = training(X_tr,T_tr)                # training the single hidden layer perceptron
    print("Model built for digit ",digit, "is ready")
